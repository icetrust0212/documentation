## Try it out

The code below shows how to login to the Verida network using the verida connect which uses wallet connect an optional configuration to automatically establish a wallet connection upon sign in .
 It is live
 - You can press the "Login" button to login, and edit the code to see how it works
 - After a successfully sign in you can sign a personal message using the "Sign Personal Message" button

```jsx live
VeridaLoginExample = () => {
	const [veridaContext, setVeridaContext] = useState(undefined);
	const [walletAddress, setWalletAddress] = useState('');
	const [message, setMessage] = useState('');
	const [walletConnector, setWalletConnector] = useState(undefined);
	this.CONTEXT_NAME = 'Verida Tutorial';

	React.useEffect(async () => {
		// Runs after the first render() lifecycle
		// We use this to check if we are logged in

		// veridaContext is a Context object (See links below)
		if (!veridaContext) {
			// we don't have veridaContext in local state

			// Check if we have a stored session
			//  hasSession is from the package "@verida/account-web-vault"
			//  see below fod links to documentation
			if (hasSession(this.CONTEXT_NAME)) {
				// we know we have a session already
				login(); // when logged in this will just setup a Verida Context
			}
		}
	}, []);

	login = async function () {
		// Create a VaultAccount. This takes a VaultAccountConfig parameter.
		// See links below to docs on these
		const bridgeURL = 'https://bridge.walletconnect.org';
		const DEFAULT_CHAIN_ID = 'eip155:1';

		const connector = new WalletConnect.default({
			bridge: bridgeURL
		});

	    connector.on("disconnect", (error, payload) => {
          if (error) {
          console.error(error);
          return;
        }
        console.log("WalletConnect on disconnect payload:", payload);
        });

		connector.on("connect", (error, payload) => {
	      console.log("WalletConnect on connect payload:", payload);
        })

		const account = new VaultAccount({
			request: {
				logoUrl: 'https://developers.verida.io/img/tutorial_login_request_logo_170x170.png'
			},
			walletConnect: {
				version: connector.version,
				uri: connector.uri,
				chainId: DEFAULT_CHAIN_ID
			}
		});

		console.log(account)

		const context = await Network.connect({
			client: {
				environment: 'testnet'
			},
			account: account,
			context: {
				name: this.CONTEXT_NAME
			}
		});

		if (context) {
			// set the local state variable
			setVeridaContext(context);
		}
	};

	logout = async function () {
		// disconnect the Verida session
		await veridaContext.account.disconnect(this.CONTEXT_NAME);
		// reset the internal state
		setVeridaContext(undefined);
	};

	convertUtf8ToHex = function(str) {
      return Array.from(str).map(c => 
	  c.charCodeAt(0) < 128 ? c.charCodeAt(0).toString(16).padStart(2, '0') :
	  encodeURIComponent(c).replace(/\%/g,'').toLowerCase()
      ).join('');
    }

	hexToUtf8 = function(hex) {
        return decodeURIComponent('%' + hex.match(/.{1,2}/g).join('%'));
    }

	signInPersonalMessage = async function(){	
       // const message = "Signing a Personal message from " + this.CONTEXT_NAME;
       // const msgParams = [
       // this.convertUtf8ToHex(message),
       // walletAddress
	   //];
	   //const result = await walletConnector.signPersonalMessage(msgParams)
	   //setMessage(this.hexToUtf8)
		//console.log(result)
		console.log('hello');
	}

	this.login = this.login.bind(this);
	this.logout = this.logout.bind(this);
	this.convertUtf8ToHex = this.convertUtf8ToHex.bind(this);
	this.hexToUtf8 = this.hexToUtf8.bind(this);
	this.signInPersonalMessage = this.signInPersonalMessage.bind(this);

	// if we have a veridaContext we are logged in

	if (veridaContext) {
		// user is logged in
		return (
			<div>
				<h3>
					Logged in! Your <a href='/docs/concepts/accounts-and-identity'>DID</a>
					is: <pre>{veridaContext.account.accountDid}</pre>
				</h3>
				<h3>
					Personal Message signed 
					 <pre>{message}</pre>
				</h3>
				<button onClick={this.logout}>Logout</button>
				<button onClick={this.signInPersonalMessage}>Sign Personal Message</button>
			</div>
		);
	} else {
		return (
			<div>
				<button onClick={this.login}>Login</button>
			</div>
		);
	}
}
```
